AJAX原生

  • 异步加载必须先实例化一个XMLHttpRequest函数。
let xhr = new XMLHttpRequest();
xhr.open();
xhr.send();
setRequsetHeader();
  • XMLHttpRequest的响应方法
/*
    readyStateChange 用于监听readyState的变化
    status 响应状态码 1xx 2xx 3xx 4xx 5xx
    statusText 响应状态说明
    readyState    请求／响应过程的当前活动状态
        0
        1
        2    发送，已经调用send方法
        3    接收响应数据
        4    表示完成

    responseText    响应主体返回的文本
    responseXML        响应主体返回的XML

*/
  • GET 和 POST 的一般写法
// get
let xhr=new XMLHttpRequest();
url="http://www.abc.com?a=b&c=d";
xhr.open("get",url,true);
xhr.send(null);
xhr.readyStateChange=function(){
  if(xhr.readyState==4&&xhr.status==200){
    console.log(xhr.responseText);
  }else{
    console.log(`error${xhr.status}`);
  }
};

//post
let xhr=new XMLHttpRequest();
url="http://abc.com";
data="a=b&c=d";
xhr.open("post",url,true);
xhr.setRequestHeader("Content-Type","application/x-www-from-urlencoded");
xhr.send(data);
xhr.readyStateChange=function(){
  if(xhr.readyState==4&&xhr.status=200){
    console.log(xhr.responseText);
  }else{
    console.log(`error:${xhr.status}`);

  }
}

创建 RegExp

//1.
let patt=/abc/g;
//2.

修饰符

/*

    i    不区分大小
    g    全局匹配
    m    多行匹配
    u    （es6新增）

*/

元字符 . \b \w\d\s

/*
    .          匹配任意字符
    \b         匹配单词边界
    \w        匹配数字，字母，下划线
    \d        匹配数字
    \s        匹配空格
    [abc]    匹配方括号里任意一个字符
    ^        匹配首部
    $        匹配尾部

    --取反--
    \B \W \D \S [^abc]

*/

量词 * + ? {}

/*

    *        匹配任意个
    +        匹配一到多个
    ?        匹配 0 到 1 个
    {n}        匹配 n 个
    {n,}    匹配至少 n 个
    {n,m}    匹配 n 到 m 个

*/

分枝条件 |

/*

a|b         匹配 a 或者 b 

*/

分组 ()

/*

    (exp)            匹配且分组
    (?<name>exp)    匹配且以name名分组
    (?:exp)            只匹配不捕获不分组
    (?=exp)            匹配exp前面位置
    (?<=exp)        匹配exp后面位置
    (?!exp)            匹配后面不跟着exp的位置
    (?<!exp)        匹配前面不跟着exp的位置
    (?#comment)        注释

 */

懒惰限定符

/*

    *?
    +?
    ??
    {n,m}?
    {n,}?

*/

转义字符 \

/*

    需要转义的字符   * . ? + $ ^ [ ] ( ) { } / \ |

*/

事件：用户或者浏览器自身执行的某种动作

事件处理程序（事件监听器）：响应某个事件的函数，事件处理程序的名字以“on”开头。

DOM0级：将一个函数赋值给一个事件处理程序属性

btn.onclick=function(){
}

DOM1级

DOM2级事件处理程序

/*
*1参数为要处理的事件名
*2参数事件处理程序的函数
*3参数为布尔值，true表示捕获阶段调用2，false表示冒泡阶段调用2
    fn在当前作用于运行
*/
addEventListener("",fn,boolean)
/*
参数同上，只是调用变成删除
*/
removeEventListener()

捕获：从不具体的父元素逐步传到具体的元素

冒泡：具体的元素逐步往上传播到不具体的父元素

DOM事件流（DOM2）:分为事件捕获阶段，处于目标阶段和事件冒泡阶段。（但浏览器都支持捕获和冒泡阶段对目标对象进行操作）

事件处理程序分类

HTML事件处理程序 ：写在html标签里

IE事件处理程序：

// fn会在全局作用域中运行
attachEvent()
detachEvent()

webpack基础应用

发表于 2016-12-22 | 分类于 工具

官网：https://webpack.js.org/

webpack是一款模块加载和打包的工具

1.全局安装

 npm install -g webpack

进入你的项目文件夹（比如叫yourproject）和 初始化

cd yourproject
npm init

初始化用来生成package.json包

{
  "name": "webpacktest",
  "version": "1.0.0",
  "description": "",
  "main": "index.js",
  "scripts": {
    "test": "echo \"Error: no test specified\" && exit 1"
  },
  "author": "",
  "license": "ISC",
  "devDependencies": {
    "webpack": "^1.14.0"
  }
}

安装依赖

npm install webpack --save-dev

生成mode_modules文件夹以及在package.json的devDependencies中自动添加

"devDependencies": {
    "webpack": "^1.14.0"
  }

在每个项目根目录下必须配置一个webpack.config.js文件，此文件用来写配置项，用来告诉webpack做什么。
webpack.config.js里一般写法

//var webpack=require('webpack');
//var commonPlugin = new webpack.optimize.CommonsChunkPlugin('comon.js');


module.explorts={
  //plugins:[commonPlugin],
  //entry里为设置处理的文件路径入口
  entry:{
    index:'./src/js/page/index.js'
    },
  //output里为处理后存放的文件夹以及新的文件名
  output:{
    path:'dist/js/page',
    //publiPath:'',
    filename:'[name].js'
    },
  // module这里定义了处理指定后缀文件的插件，插件需要安装在项目内
  // npm install 插件名 -dev--save
  module:{
    loaders:[
      //处理css和scss文件
        {test:/\.css$/,
         loader:'style-loader!css-loader!sass-loader'
        },
      //处理es6文件
        {
          test:/\.js$/,
          //排除不需要解析道文件夹
          exclude:(),
          loader:'label-loader',
          //
          qurey:{
            presets:['es2015']
          }
        }

      ]
    },
  //
  resolve:{
    // 设置查找module的根目录，可忽略
    root:'',
    // resolve.extensions设置扩展文件后缀名， require()可以省略文件名
    extensions:['','js','css'],
    //设置模块别名
    alias:{

      }
    },
  //插件
  plugins:{

  }
}
